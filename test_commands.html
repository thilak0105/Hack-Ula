Test Commands - Copy and Paste into Chrome DevTools

Step 1: Check Available Models

window.Android.getAvailableModels(function(models) {
    console.log('=== AVAILABLE MODELS ===');
    console.log('Total models:', models.length);
    models.forEach(function(m, i) {
        console.log('\nModel', i + 1);
        console.log('  Name:', m.name);
        console.log('  ID:', m.id);
        console.log('  Size:', m.sizeInMB, 'MB');
        console.log('  Downloaded:', m.isDownloaded ? '✅ YES' : '❌ NO');
    });
});

Step 2: Download TinyLlama Model

window.Android.downloadModel(
    'tinyllama-1.1b',
    function(progress) {
        console.log('⬇️ Download:', progress + '%');
    },
    function(success) {
        console.log(success ? '✅ COMPLETE!' : '❌ FAILED');
        // Verify after download
        window.Android.getAvailableModels(function(models) {
            models.forEach(function(m) {
                console.log(m.name + ':', m.isDownloaded ? '✅' : '❌');
            });
        });
    }
);

Step 3: Test Model Loading

window.Android.loadModel('tinyllama-1.1b', function(success) {
    console.log('Load Model Result:', success ? '✅ SUCCESS' : '❌ FAILED');
});

Step 4: Check if Model is Loaded

window.Android.isModelLoaded(function(isLoaded) {
    console.log('Model Loaded?', isLoaded ? '✅ YES' : '❌ NO');
});

Step 5: Test Simple Text Generation

window.Android.generateText('Hello, how are you?', function(response) {

});

Step 6: Test Course Generation (After Model Downloaded)

window.Android.generateCourseContent(
    '',
    '',
    'Test Course',
    'Beginner',
    'Students',
    'None',
    function(response) {
        console.log('Success:', response.success);
        console.log('Message:', response.message);
        if (response.success) {
            console.log('Course:', response.course.title);
            console.log('Modules:', response.course.modules ? response.course.modules.length : 0);
        } else {
            console.error('Error:', response.error);
        }
    }
);

Step 7: Test Lesson Content Generation

window.Android.generateLessonContent(

            console.error('Error:', response.error);
